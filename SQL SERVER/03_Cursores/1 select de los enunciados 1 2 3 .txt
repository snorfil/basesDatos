SELECT e.nomempleado, pc.idpedido, SUM(pd.preciounidad * pd.cantidad)
FROM rrhh.empleados e
join venta.pedidoscabe pc ON e.idempleado = pc.idempleado 
join PEDIDOSDETA pd ON pc.idpedido = pd.idpedido 
GROUP BY e.nomempleado, pc.idpedido
ORDER BY 1,2

FROM venta.PEDIDOSCABE C
JOIN  PEDIDOSDETA D ON C.IDPEDIDO=D.IDPEDIDO
JOIN rrhh.empleados e ON C.idempleado=e.idempleado
GROUP BY YEAR(FECHAPEDIDO), C.idempleado, e.nomempleado
ORDER BY 1,

SELECT P.nomproducto,C.idpedido,C.fechapedido,
D.preciounidad*D.cantidad
FROM venta.PEDIDOSCABE C
JOIN  PEDIDOSDETA D ON C.IDPEDIDO=D.IDPEDIDO
JOIN compra.productos P ON D.idproducto=P.idproducto
where YEAR(FECHAPEDIDO)=2012
ORDER BY 1,2
